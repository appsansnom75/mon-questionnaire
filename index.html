<script>

const QUESTIONS_URL = "https://xlncdpwghwqorjgkuecu.supabase.co"; 

const WORKER_URL = "https://compatibility-worker.appsansnom.workers.dev/";   

const MY_PROFILE = { securite: 0.7, autonomie: 0.4 }; 


let questions = [];

let asked = [];

let answers = [];

let scores = {};


fetch(QUESTIONS_URL)

  .then(r => r.json())

  .then(q => { questions = q; nextQuestion(); });


// Vérifie si la question peut être posée

function eligible(q) {

  if (asked.includes(q.id)) return false;

  if (!q.trigger_rule || q.trigger_rule === "always") return true;

  try {

    const expr = q.trigger_rule.replace(/\b(\w+)\b/g, (_, d) => `(scores["${d}"]||0)`);

    return eval(expr);

  } catch(e) {

    console.error("Erreur trigger_rule question", q.id, e);

    return false;

  }

}


// Affiche la question et génère les réponses dynamiquement

function renderQuestion() {

  const container = document.getElementById("question");

  container.innerHTML = "";


  const qText = document.createElement("p");

  qText.innerText = current.text;

  container.appendChild(qText);


  const mapping = current.response_score_mapping || [];


  // scale ou yesno ou choices

  if(current.type === "scale") {

    const labels = ["Pas du tout","Un peu","Beaucoup"];

    labels.forEach((label, i) => {

      const score = mapping[i] !== undefined ? mapping[i] : i;

      const btn = document.createElement("button");

      btn.innerText = label;

      btn.onclick = () => recordAnswer(score);

      container.appendChild(btn);

    });

  } 

  else if(current.type === "yesno") {

    const labels = current.options || ["Oui","Non"];

    labels.forEach((label,i) => {

      const score = mapping[i] !== undefined ? mapping[i] : i;

      const btn = document.createElement("button");

      btn.innerText = label;

      btn.onclick = () => recordAnswer(score);

      container.appendChild(btn);

    });

  } 

  else if(current.type === "choices" && current.options) {

    current.options.forEach((label,i) => {

      const score = mapping[i] !== undefined ? mapping[i] : i;

      const btn = document.createElement("button");

      btn.innerText = label;

      btn.onclick = () => recordAnswer(score);

      container.appendChild(btn);

    });

  } 

  else if(current.type === "text") {

    const input = document.createElement("input");

    input.type = "text";

    input.id = "textAnswer";

    const btn = document.createElement("button");

    btn.innerText = "Valider";

    btn.onclick = () => recordAnswer(0); // texte libre = score 0

    container.appendChild(input);

    container.appendChild(btn);

  }

}


// Enregistre la réponse

function recordAnswer(val) {

  answers.push({

    dimension: current.dimension,

    value: val,

    weight: current.weight || 1

  });

  scores[current.dimension] = (scores[current.dimension] || 0) + val * (current.weight || 1);

  nextQuestion();

}


// Prochaine question

function nextQuestion() {

  if (asked.length >= 20) return finish();

  const q = questions.find(eligible);

  if (!q) return finish();

  asked.push(q.id);

  window.current = q;

  renderQuestion();

}


// Fin du questionnaire

function finish() {

  document.getElementById("title").innerText = "Calcul en cours…";

  document.getElementById("question").innerText = "";


  fetch(WORKER_URL, {

    method: "POST",

    body: JSON.stringify({ action: "profile", answers })

  })

  .then(r => r.json())

  .then(profile => {

    fetch(WORKER_URL, {

      method: "POST",

      body: JSON.stringify({

        action: "compatibility",

        me: MY_PROFILE,

        friend: profile

      })

    })

    .then(r => r.json())

    .then(res => {

      document.body.innerHTML = `<h2>Compatibilité : ${res.percent}%</h2>

        <div id="explanation">Chargement de l'explication...</div>`;


      fetch("https://api.openai.com/v1/chat/completions", {

        method: "POST",

        headers: {

          "Content-Type": "application/json",

          "Authorization": "Bearer TON_OPENAI_API_KEY"

        },

        body: JSON.stringify({

          model: "gpt-4",

          messages: [{

            role: "user",

            content: `

Explique de manière simple et bienveillante

pourquoi deux personnes ont une compatibilité de ${res.percent} %.


Profil A : ${JSON.stringify(MY_PROFILE)}

Profil B : ${JSON.stringify(profile)}

`

          }],

          max_tokens: 200

        })

      })

      .then(r => r.json())

      .then(data => {

        const explanation = data.choices[0].message.content;

        document.getElementById("explanation").innerText = explanation;

      })

      .catch(e => {

        console.error(e);

        document.getElementById("explanation").innerText = "Impossible de charger l'explication IA.";

      });

    });

  });

}

</script>
